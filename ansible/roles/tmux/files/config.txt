
# Bindings
set -g prefix C-s

unbind r
bind r source-file ~/.tmux.conf
bind-key w swap-pane
bind-key C-q run-shell "tmux kill-session"

# Copy
# sudo apt install xclip
bind -T copy-mode-vi y send-keys -X copy-pipe-and-cancel "xclip -in -selection clipboard"
bind p run-shell "xclip -out -selection clipboard | tmux load-buffer - && tmux paste-buffer"

# cosmetic
set -g status-position top

# Plugins
# git clone https://github.com/tmux-plugins/tpm ~/.tmux/plugins/tpm
set -g @plugin "tmux-plugins/tpm"
set -g @plugin "tmux-plugins/tmux-online-status"

set -g @plugin 'tmux-plugins/tmux-resurrect'
set -g @plugin 'tmux-plugins/tmux-continuum'

set -g @plugin 'tmux-plugins/tmux-sensible'

set -g @plugin 'christoomey/vim-tmux-navigator'

# Make sure that the colors are the same in and out of tmux
set-option -sa terminal-overrides ",xterm*:Tc"

set -g @catppuccin_flavor 'mocha' # latte, frappe, macchiato or mocha

set -g @catppuccin_window_status_style "none"
set -g @catppuccin_window_left_separator "none"
set -g @catppuccin_window_middle_separator "none"
set -g @catppuccin_window_right_separator "none"

# Plugin: continuum
set -g @continuum-restore 'on'
set -g @continuum-boot-options 'kitty'
#set -g @resurrect-strategy-vim 'session'
set -g @resurrect-capture-pane-contents 'on'

# Initialize TMUX plugin manager (keep this line at the very bottom of tmux.conf)
run -b '~/.tmux/plugins/tpm/tpm'
